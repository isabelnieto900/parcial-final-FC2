16double IntegradorMonteCarlo::CalcularIntegralSimple(long long numero_muestras, double& error_estimado) {
17 if (numero_muestras <= 0) {
18 error_estimado = 0.0; // O un valor indicativo de error como NaN o infinito
19 // std::cerr << "Advertencia: numero_muestras debe ser positivo." << std::endl;
20 return 0.0; // O NaN
21 }
22
23 double suma_f = 0.0;
24 double suma_f_cuadrado = 0.0;
25
26 for (long long i = 0; i < numero_muestras; ++i) {
27 double x_aleatorio = distribucion_uniforme(gen); // Genera x_i uniformemente en [a, b)
28 double valor_f_xi = func_a_integrar(x_aleatorio);
29 suma_f += valor_f_xi;
30 suma_f_cuadrado += valor_f_xi * valor_f_xi;
31 }
32
33 // Estimación de la integral I = (b-a) * <f>
34 // donde <f> es el promedio de f(x_i)
35 double promedio_f = suma_f / static_cast<double>(numero_muestras);
36 double valor_integral = (limite_superior - limite_inferior) * promedio_f;
37
38 // Estimación del error (error estándar de la media de la integral)
39 // Error = (b-a) * sqrt( ( <f^2> - <f>^2 ) / N )
40 // = (b-a) * sigma_f / sqrt(N)
41 if (numero_muestras > 1) {
42 double promedio_f_cuadrado = suma_f_cuadrado / static_cast<double>(numero_muestras);
43 double varianza_f = promedio_f_cuadrado - (promedio_f * promedio_f);
44
45 // La varianza no puede ser negativa, esto puede ocurrir por errores de precisión numérica.
46 if (varianza_f < 0) {
47 varianza_f = 0.0;
48 }
49 error_estimado = (limite_superior - limite_inferior) * std::sqrt(varianza_f / static_cast<double>(numero_muestras));
50 } else {
51 // No se puede estimar la varianza (y por lo tanto el error) con una sola muestra.
52 // Se podría devolver NaN o un valor grande para indicar esto.
53 error_estimado = 0.0; // O std::numeric_limits<double>::quiet_NaN(); si se incluye <limits>
54 }
55
56 return valor_integral;
57}
